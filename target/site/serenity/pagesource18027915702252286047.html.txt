function validateEmail(email) {
    var re = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    return re.test(String(email).toLowerCase());
}

                case "kg":
                    customItemGondolaProduct = customItemGondolaProduct.replace(/%pesoproduct%/g, item.UnidadPrecio + " kg");
                    break;
            }

            itemproductTopTen += customItemGondolaProduct;
            indice++;
            if (indice == cantitems) {                
                templateGondolaProduct = templateGondolaProduct + '<div class="group-item"><ul>' + itemproductTopTen + '</ul></div>';
                indice = 0;
                itemproductTopTen = '';                                
            }
        });

        if (indice > 0 && indice != cantitems) {
            templateGondolaProduct = templateGondolaProduct + '<div class="group-item"><ul>' + itemproductTopTen + '</ul></div>';

        }
        templateGondolaProduct = templateGondolaProduct + '</div></div></div></div></section>';
        templateGondolaProduct = templateGondolaProduct + '<div>%divButtonViewMore%</div>';
        if (linkViewMore!='' && bdwMobile.any()){
          templateGondolaProduct = templateGondolaProduct.replace('%divButtonViewMore%', '<div class="row events-category-box-bottom" style="border-radius: 5px;text-align:center;background:#ec1c24;color:#fff;width: 80%;margin: auto;padding: 5px;margin-bottom:10px"><span style=""><a style="color:#fff" href="' + linkViewMore + '">Ver más</a></span></div>');      
          templateGondolaProduct = templateGondolaProduct.replace(/%sectionEvent%/g, eventName); 
        }
        else
        {
          templateGondolaProduct = templateGondolaProduct.replace('%divButtonViewMore%','');
        }
        templateGondolaProduct += braindw_style_dynamic();
        if(configs != undefined && configs != null && configs != "" && bdwMobile.any() == null){
            templateGondolaProduct= templateGondolaProduct.replace(/%left-percent%/g,configs.controlsStylePrev);
            templateGondolaProduct= templateGondolaProduct.replace(/%right-percent%/g,configs.controlsStyleNext);
            templateGondolaProduct= templateGondolaProduct.replace(/%braindw_id_gondola%/g,braindw_id_gondola);
        }
        
        return templateGondolaProduct;
    }
    return "";
}
